Add Epoch Time
	[Arguments]  ${variable}  ${date}
	Log  ${date}
	${epoch} =   Convert Date  ${date}  epoch
	${num} =     Convert To Integer     ${epoch}
	${time} =    Evaluate   ${num}*1000
	${path} =    Evaluate  '${path}' + '&' + '${variable}' + '=' +'${time}'
	Set Suite Variable   ${path}
